// melee
func MeleeSword(swordExtra, shieldExtra)
  equipL sword *10 + swordExtra
  equipR shield *10 + shieldExtra

func MeleeHammer(hammerExtra, shieldExtra)
  equipL hammer *10 + hammerExtra
  equipR shield *10 + shieldExtra

func MeleeDPS(swordExtra, hammerExtra)
  equipL sword *10 + swordExtra
  equipR hammer *10 + hammerExtra

// ranged
func Ranged(shieldExtra)
  equipL crossbow *10
  equipR shield *10 + shieldExtra

func RangedRepeat()
  equip repeating crossbow *10

func RangedHealing()
  equipL crossbow *10
  equipR ouroboros

func RangedWand()
  equipL crossbow *10
  equipR wand *10


// magic
func Staff(staffExtra)
  equip staff *10 + staffExtra


// special
func Dash(swordExtra)
  equipL sword *10 + swordExtra
  equipR shield *10 dashing

func LostItem(lostItem)
  equip lostItem *10


// out of battle
func healstvr()
  ?pickup.distance < 10
    equipL star
    equipR shield *10
  :?foe.distance > 20
    ?hp < maxhp
      equipL ouroboros
      equipR shield *10
    :
      equipL triskelion
      equipR shield *10
